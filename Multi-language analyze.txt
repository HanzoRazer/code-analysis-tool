+from __future__ import annotations
+
+import os
+import platform
+from dataclasses import dataclass
+from pathlib import Path
+from typing import Optional
+
+
+class TreeSitterUnavailable(RuntimeError):
+    pass
+
+
+@dataclass(frozen=True)
+class TreeSitterConfig:
+    """Configuration for loading/building Tree-sitter language libraries.
+
+    This is intentionally deterministic:
+    - Uses vendored grammar sources when present.
+    - Produces a single shared library in a stable cache location.
+    - Never downloads grammars at runtime.
+    """
+
+    vendor_root: Path
+    cache_dir: Path
+    library_name: str = "code_audit_treesitter_langs"
+
+    @staticmethod
+    def default() -> "TreeSitterConfig":
+        pkg_root = Path(__file__).resolve().parents[2]
+        vendor_root = pkg_root / "data" / "treesitter" / "vendor"
+        cache_dir = Path(os.environ.get("CODE_AUDIT_TS_CACHE", "")) or (Path.home() / ".cache" / "code_audit")
+        return TreeSitterConfig(vendor_root=vendor_root, cache_dir=cache_dir)
+
+
+def _lib_suffix() -> str:
+    sys = platform.system().lower()
+    if sys == "darwin":
+        return ".dylib"
+    if sys == "windows":
+        return ".dll"
+    return ".so"
+
+
+def _import_tree_sitter():
+    try:
+        from tree_sitter import Language, Parser  # type: ignore
+    except Exception as e:  # pragma: no cover
+        raise TreeSitterUnavailable(
+            "tree-sitter is not installed. Install extras: pip install .[treesitter]"
+        ) from e
+    return Language, Parser
+
+
+def language_library_path(cfg: Optional[TreeSitterConfig] = None) -> Path:
+    cfg = cfg or TreeSitterConfig.default()
+    cfg.cache_dir.mkdir(parents=True, exist_ok=True)
+    return (cfg.cache_dir / (cfg.library_name + _lib_suffix())).resolve()
+
+
+def vendored_grammar_dirs(cfg: Optional[TreeSitterConfig] = None) -> list[Path]:
+    """Return vendored grammar directories if present.
+
+    Expected layout:
+      vendor/tree-sitter-javascript/
+      vendor/tree-sitter-typescript/typescript/
+      vendor/tree-sitter-typescript/tsx/
+    """
+    cfg = cfg or TreeSitterConfig.default()
+    v = cfg.vendor_root.resolve()
+    candidates = [
+        v / "tree-sitter-javascript",
+        v / "tree-sitter-typescript" / "typescript",
+        v / "tree-sitter-typescript" / "tsx",
+    ]
+    return [p for p in candidates if p.exists() and p.is_dir()]
+
+
+def build_language_library(cfg: Optional[TreeSitterConfig] = None) -> Path:
+    """Build the shared language library from vendored grammars.
+
+    Raises TreeSitterUnavailable if vendored grammars are missing.
+    """
+    cfg = cfg or TreeSitterConfig.default()
+    Language, _Parser = _import_tree_sitter()
+    out = language_library_path(cfg)
+    grammars = vendored_grammar_dirs(cfg)
+    if len(grammars) < 3:
+        raise TreeSitterUnavailable(
+            "Vendored tree-sitter grammars are missing. Expected:\n"
+            f"  {cfg.vendor_root}/tree-sitter-javascript\n"
+            f"  {cfg.vendor_root}/tree-sitter-typescript/typescript\n"
+            f"  {cfg.vendor_root}/tree-sitter-typescript/tsx\n"
+            "See src/code_audit/data/treesitter/vendor/README.md"
+        )
+
+    # Deterministic build: input is vendored dirs, output is fixed file path.
+    # Note: tree_sitter.Language.build_library compiles grammar C sources.
+    Language.build_library(str(out), [str(p) for p in grammars])
+    return out
+
