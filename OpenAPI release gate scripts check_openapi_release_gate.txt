+from __future__ import annotations
+
+import json
+import sys
+from pathlib import Path
+
+try:
+    import jsonschema
+except Exception as e:  # pragma: no cover
+    raise RuntimeError(
+        "jsonschema is required to validate BOM generator gate result output."
+    ) from e
+
+
+ROOT = Path(__file__).resolve().parents[1]
+SCHEMA = ROOT / "schemas" / "release_bom_generator_gate_result.schema.json"
+
+
+def _load_json(path: Path):
+    return json.loads(path.read_text(encoding="utf-8"))
+
+
+def main(argv: list[str]) -> int:
+    if len(argv) != 2:
+        print("Usage: python scripts/validate_release_bom_generator_gate_result.py <result.json>", file=sys.stderr)
+        return 2
+
+    target = ROOT / argv[1]
+    if not SCHEMA.exists():
+        print(f"Missing schema: {SCHEMA.relative_to(ROOT)}", file=sys.stderr)
+        return 1
+    if not target.exists():
+        print(f"Missing file: {target.relative_to(ROOT)}", file=sys.stderr)
+        return 1
+
+    schema = _load_json(SCHEMA)
+    obj = _load_json(target)
+
+    v = jsonschema.Draft202012Validator(schema)
+    errs = sorted(v.iter_errors(obj), key=lambda e: list(e.path))
+    if errs:
+        print("[release-bom-generator-gate-result-schema] FAIL", file=sys.stderr)
+        for e in errs[:50]:
+            path = ".".join([str(p) for p in e.path]) or "<root>"
+            print(f"  - {path}: {e.message}", file=sys.stderr)
+        if len(errs) > 50:
+            print(f"  ... ({len(errs) - 50} more)", file=sys.stderr)
+        return 1
+
+    print("[release-bom-generator-gate-result-schema] OK")
+    return 0
+
+

+# Stable enum tokens (must stay aligned with the BOM schema enum).
+EDGE_KIND_KEYS: Set[str] = {
+    "direct",
+    "allOf", "anyOf", "oneOf",
+    "not", "if", "then", "else",
+    "items", "additionalProperties",
+    "properties", "patternProperties",
+    "dependentSchemas",
+    "definitions", "$defs",
+    "unknown",
+}
+
+
+# “Most specific wins” priority table.
+# Smaller number = more specific = wins.
+#
+# Rationale:
+# - Structural placement like properties/items is more semantically specific than combinators like allOf/anyOf.
+# - Conditional branches (if/then/else) are also specific.
+# - $defs/definitions are definitional containers (rarely where $ref should appear directly; still classified).
+# - direct is only when no container context exists (or none recognized).
+KIND_PRIORITY: Dict[str, int] = {
+    # leaf-like / strongest specificity
+    "properties": 10,
+    "patternProperties": 12,
+    "dependentSchemas": 14,
+    "items": 20,
+    "additionalProperties": 22,
+
+    # conditional constructs
+    "if": 30,
+    "then": 32,
+    "else": 34,
+    "not": 36,
+
+    # combinators (broader)
+    "allOf": 50,
+    "oneOf": 52,
+    "anyOf": 54,
+
+    # definitional containers (broad)
+    "$defs": 80,
+    "definitions": 82,
+
+    # base / fallback
+    "direct": 900,
+    "unknown": 1000,
+}
+
+
+def normalize_kind(key: Optional[str]) -> str:
+    if key is None:
+        return "direct"
+    if key in KIND_PRIORITY:
+        return key
+    return "unknown"
+
+
+def choose_most_specific_kind(context_keys: List[Optional[str]]) -> str:
+    """
+    Given a stack of parent keys (outer -> inner), choose the most specific kind
+    by numeric priority. Tie-breaker: deterministic by lexical key name.
+    """
+    kinds = [normalize_kind(k) for k in context_keys]
+    # pick min priority; on tie, min lexical to avoid nondeterminism
+    best = min(kinds, key=lambda k: (KIND_PRIORITY.get(k, 1000), k))
+    return best
+
+
+def collect_ref_targets_with_context(node: Any) -> List[Tuple[str, str]]:
+    """
+    Returns list of (ref, kind) pairs discovered in the schema.
+    For each $ref, kind is chosen by “most specific wins” across all nested container keys.
+    """
+    out: List[Tuple[str, str]] = []
+
+    def walk(n: Any, stack: List[Optional[str]]) -> None:
+        if isinstance(n, dict):
+            ref = n.get("$ref")
+            if isinstance(ref, str):
+                kind = choose_most_specific_kind(stack)
+                out.append((ref, kind))
+            for k, v in n.items():
+                # push this container key; it's part of context for nested schemas
+                walk(v, stack + [k])
+        elif isinstance(n, list):
+            for it in n:
+                walk(it, stack)
+
+    walk(node, [])
+    return out
