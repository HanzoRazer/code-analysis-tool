#!/usr/bin/env python3
"""
Code Quality Analyzer Suite
A comprehensive tool for analyzing code quality issues in JavaScript/Vue projects.
"""

import os
import sys
import json
import argparse
import ast
import re
from pathlib import Path
from datetime import datetime
from typing import Dict, List, Any, Optional
import glob

class CodeQualityAnalyzer:
    """Main orchestrator for all code quality checks"""
    
    def __init__(self, path: str, verbose: bool = False, json_output: bool = False):
        self.path = Path(path)
        self.verbose = verbose
        self.json_output = json_output
        self.results = {
            "summary": {
                "total_files": 0,
                "total_issues": 0,
                "critical_issues": 0,
                "warning_issues": 0,
                "info_issues": 0
            },
            "checks": {},
            "timestamp": datetime.now().isoformat(),
            "target_path": str(path)
        }
        self.issues_found = []
        
    def log(self, message: str, level: str = "info"):
        """Logging utility"""
        if self.verbose or level == "error":
            print(f"[{level.upper()}] {message}")
    
    def add_issue(self, check_name: str, file_path: str, line: int, 
                  message: str, severity: str = "warning", suggestion: str = ""):
        """Add an issue to results"""
        issue = {
            "check": check_name,
            "file": str(file_path),
            "line": line,
            "message": message,
            "severity": severity,
            "suggestion": suggestion
        }
        self.issues_found.append(issue)
        
        # Update summary counts
        if severity == "critical":
            self.results["summary"]["critical_issues"] += 1
        elif severity == "warning":
            self.results["summary"]["warning_issues"] += 1
        else:
            self.results["summary"]["info_issues"] += 1
        
        self.results["summary"]["total_issues"] += 1
    
    def analyze(self):
        """Run all analysis checks"""
        self.log("Starting code quality analysis...")
        
        # Find all relevant files
        files = self._get_files_to_analyze()
        self.results["summary"]["total_files"] = len(files)
        self.log(f"Found {len(files)} files to analyze")
        
        # Run all checks
        checks = [
            RecursiveLinesDetector(self),
            CyclicDependencyDetector(self),
            ComponentDepthAnalyzer(self),
            GodComponentDetector(self),
            PropsDrillingAnalyzer(self),
            CallbackHellDetector(self),
            MemoryLeakPatternDetector(self),
            LongParameterListDetector(self),
            DuplicateCodeDetector(self),
            MagicNumberDetector(self),
            DeepNestingDetector(self),
            SecurityVulnerabilityDetector(self),
            MissingKeyPropsDetector(self),
            PropMutationDetector(self),
            MissingDependencyArrayDetector(self),
            InlineFunctionDetector(self),
            TodoCommentDetector(self),
            DeadCodeDetector(self),
            CircularComponentDetector(self),
            BundleSizeAnalyzer(self),
            TestCoverageIndicator(self)
        ]
        
        for check in checks:
            check_name = check.__class__.__name__
            self.results["checks"][check_name] = {"issues": 0}
            self.log(f"Running {check_name}...")
            
            try:
                check.run(files)
                issues_count = len([i for i in self.issues_found if i["check"] == check_name])
                self.results["checks"][check_name]["issues"] = issues_count
                self.results["checks"][check_name]["status"] = "completed"
            except Exception as e:
                self.log(f"Error in {check_name}: {str(e)}", "error")
                self.results["checks"][check_name]["status"] = f"failed: {str(e)}"
        
        self.results["issues"] = self.issues_found
        
        # Output results
        if self.json_output:
            print(json.dumps(self.results, indent=2))
        else:
            self._print_human_readable()
    
    def _get_files_to_analyze(self) -> List[Path]:
        """Get all relevant files for analysis"""
        patterns = [
            "**/*.js", "**/*.jsx", "**/*.ts", "**/*.tsx", 
            "**/*.vue", "**/*.html", "**/*.css"
        ]
        
        files = []
        for pattern in patterns:
            files.extend(self.path.glob(pattern))
        
        # Exclude common directories
        exclude_dirs = {"node_modules", "dist", "build", ".git", "__pycache__"}
        files = [f for f in files if not any(part in exclude_dirs for part in f.parts)]
        
        return files
    
    def _print_human_readable(self):
        """Print results in human-readable format"""
        print("\n" + "="*80)
        print(f"CODE QUALITY ANALYSIS REPORT")
        print("="*80)
        print(f"Target: {self.results['target_path']}")
        print(f"Timestamp: {self.results['timestamp']}")
        print(f"Files analyzed: {self.results['summary']['total_files']}")
        print(f"Total issues: {self.results['summary']['total_issues']}")
        print(f"  Critical: {self.results['summary']['critical_issues']}")
        print(f"  Warnings: {self.results['summary']['warning_issues']}")
        print(f"  Info: {self.results['summary']['info_issues']}")
        
        if self.issues_found:
            print("\n" + "-"*80)
            print("DETAILED ISSUES:")
            print("-"*80)
            
            # Group by file for better readability
            by_file = {}
            for issue in self.issues_found:
                if issue['file'] not in by_file:
                    by_file[issue['file']] = []
                by_file[issue['file']].append(issue)
            
            for file_path, issues in by_file.items():
                print(f"\n📁 {file_path}")
                for issue in issues:
                    severity_icon = {
                        "critical": "🔴",
                        "warning": "🟡",
                        "info": "🔵"
                    }.get(issue['severity'], "⚪")
                    
                    print(f"  {severity_icon} Line {issue['line']}: {issue['message']}")
                    if issue['suggestion']:
                        print(f"     💡 Suggestion: {issue['suggestion']}")
        else:
            print("\n✅ No issues found! Your code looks clean!")

# Individual check implementations

class BaseCheck:
    """Base class for all checks"""
    
    def __init__(self, analyzer):
        self.analyzer = analyzer
    
    def run(self, files):
        """Override in child classes"""
        pass
    
    def read_file(self, file_path):
        """Safely read file contents"""
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                return f.readlines()
        except Exception as e:
            self.analyzer.log(f"Error reading {file_path}: {e}", "error")
            return []

class RecursiveLinesDetector(BaseCheck):
    """Detect recursive function calls and components"""
    
    def run(self, files):
        for file_path in files:
            if file_path.suffix in ['.js', '.ts', '.vue']:
                lines = self.read_file(file_path)
                content = ''.join(lines)
                
                # Detect direct recursion (function calling itself)
                func_pattern = r'function\s+(\w+)\s*\([^)]*\)\s*{[^}]*\1\s*\('
                for match in re.finditer(func_pattern, content):
                    line_no = self._find_line_number(lines, match.start())
                    self.analyzer.add_issue(
                        check_name=self.__class__.__name__,
                        file_path=file_path,
                        line=line_no,
                        message=f"Direct recursion detected in function '{match.group(1)}'",
                        severity="warning",
                        suggestion="Ensure there's a proper base case to prevent infinite recursion"
                    )
                
                # Detect potential infinite recursion in Vue components
                if file_path.suffix == '.vue':
                    component_name = file_path.stem
                    if f"<{component_name}" in content and f"</{component_name}>" in content:
                        # Component renders itself
                        line_no = self._find_line_number(lines, content.find(f"<{component_name}"))
                        self.analyzer.add_issue(
                            check_name=self.__class__.__name__,
                            file_path=file_path,
                            line=line_no,
                            message=f"Component '{component_name}' renders itself recursively",
                            severity="warning",
                            suggestion="Ensure recursive rendering has a stopping condition (v-if)"
                        )
    
    def _find_line_number(self, lines, position):
        """Find line number for a character position"""
        char_count = 0
        for i, line in enumerate(lines, 1):
            char_count += len(line)
            if char_count > position:
                return i
        return 1

class CyclicDependencyDetector(BaseCheck):
    """Detect circular imports/dependencies"""
    
    def run(self, files):
        imports = {}
        
        # Build import graph
        for file_path in files:
            if file_path.suffix in ['.js', '.ts', '.vue']:
                lines = self.read_file(file_path)
                imports[str(file_path)] = []
                
                for i, line in enumerate(lines, 1):
                    # Match import statements
                    import_match = re.search(r'import\s+.*\s+from\s+[\'"]([^\'"]+)[\'"]', line)
                    if import_match:
                        imported = import_match.group(1)
                        imports[str(file_path)].append((imported, i))
        
        # Detect cycles (simplified - would need proper graph algorithm)
        for file_path, deps in imports.items():
            for dep, line_no in deps:
                # Check if dep imports back to original
                dep_path = self._resolve_dep_path(file_path, dep)
                if dep_path in imports:
                    for back_dep, _ in imports[dep_path]:
                        if back_dep == file_path or back_dep in file_path:
                            self.analyzer.add_issue(
                                check_name=self.__class__.__name__,
                                file_path=file_path,
                                line=line_no,
                                message=f"Potential cyclic dependency with {dep}",
                                severity="critical",
                                suggestion="Break the circular dependency by extracting shared logic to a third module"
                            )
    
    def _resolve_dep_path(self, current_file, dep):
        """Resolve relative import paths"""
        if dep.startswith('.'):
            current_dir = os.path.dirname(current_file)
            return os.path.normpath(os.path.join(current_dir, dep))
        return dep

class ComponentDepthAnalyzer(BaseCheck):
    """Analyze component nesting depth"""
    
    def run(self, files):
        vue_files = [f for f in files if f.suffix == '.vue']
        
        for file_path in vue_files:
            lines = self.read_file(file_path)
            depth = 0
            max_depth = 0
            depth_lines = []
            
            for i, line in enumerate(lines, 1):
                # Count opening/closing tags
                opening_tags = len(re.findall(r'<[^/][^>]*>', line))
                closing_tags = len(re.findall(r'</[^>]+>', line))
                
                depth += opening_tags - closing_tags
                
                if depth > max_depth:
                    max_depth = depth
                    depth_lines = [(i, depth)]
                elif depth == max_depth:
                    depth_lines.append((i, depth))
            
            if max_depth > 5:
                for line_no, depth_val in depth_lines:
                    self.analyzer.add_issue(
                        check_name=self.__class__.__name__,
                        file_path=file_path,
                        line=line_no,
                        message=f"Deep component nesting detected (depth: {depth_val})",
                        severity="warning",
                        suggestion="Consider refactoring into smaller, composable components"
                    )

class GodComponentDetector(BaseCheck):
    """Detect overly large components (God objects)"""
    
    def run(self, files):
        for file_path in files:
            if file_path.suffix in ['.vue', '.jsx', '.tsx']:
                lines = self.read_file(file_path)
                
                # Check line count
                if len(lines) > 400:
                    self.analyzer.add_issue(
                        check_name=self.__class__.__name__,
                        file_path=file_path,
                        line=1,
                        message=f"Component is very large ({len(lines)} lines)",
                        severity="warning",
                        suggestion="Break this component into smaller, focused components"
                    )
                
                # Check method count in Vue components
                if file_path.suffix == '.vue':
                    content = ''.join(lines)
                    methods_match = re.search(r'methods\s*:\s*{([^}]+)}', content)
                    if methods_match:
                        method_count = methods_match.group(1).count('(')
                        if method_count > 15:
                            self.analyzer.add_issue(
                                check_name=self.__class__.__name__,
                                file_path=file_path,
                                line=1,
                                message=f"Component has many methods ({method_count})",
                                severity="info",
                                suggestion="Consider extracting some methods into composables/mixins"
                            )

class PropsDrillingAnalyzer(BaseCheck):
    """Detect props drilling through multiple components"""
    
    def run(self, files):
        # Track prop usage across files
        prop_passthrough = {}
        
        for file_path in files:
            if file_path.suffix == '.vue':
                lines = self.read_file(file_path)
                content = ''.join(lines)
                
                # Find props defined in component
                props_match = re.search(r'props\s*:\s*\[([^\]]+)\]', content)
                if props_match:
                    props = re.findall(r'[\'"]([^\'"]+)[\'"]', props_match.group(1))
                    
                    for prop in props:
                        # Check if prop is used only to pass to child
                        child_usage = re.findall(rf':{prop}=["\'][^"\']*["\']', content)
                        direct_usage = re.findall(rf'{prop}[^:]', content)
                        
                        if child_usage and not direct_usage:
                            prop_passthrough.setdefault(prop, []).append(str(file_path))
        
        # Report props that are passed through multiple components
        for prop, files_list in prop_passthrough.items():
            if len(files_list) > 2:
                for file_path in files_list[:-1]:  # Skip last file
                    self.analyzer.add_issue(
                        check_name=self.__class__.__name__,
                        file_path=file_path,
                        line=1,
                        message=f"Prop '{prop}' is being passed through without being used",
                        severity="info",
                        suggestion="Consider using provide/inject or Vuex for deeply nested props"
                    )

class LongParameterListDetector(BaseCheck):

        for i, line in enumerate(lines, 1):
            char_count += len(line)
            if char_count > position:
                return i
        return 1

class DuplicateCodeDetector(BaseCheck):
    """Detect duplicate code blocks (simplified)"""
    
    def run(self, files):
        # Simple duplicate line detection
        file_lines = {}
        
        for file_path in files:
            if file_path.suffix in ['.js', '.ts', '.vue']:
                lines = self.read_file(file_path)
                file_lines[str(file_path)] = lines
        
        # Compare files for similar blocks (simplified)
        for file1, lines1 in file_lines.items():
            for file2, lines2 in file_lines.items():
                if file1 >= file2:
                    continue
                
                # Look for identical blocks of 5+ lines
                for i in range(len(lines1) - 4):
                    block = ''.join(lines1[i:i+5])
                    if block in ''.join(lines2):
                        self.analyzer.add_issue(
                            check_name=self.__class__.__name__,
                            file_path=file1,
                            line=i+1,
                            message="Duplicate code block detected",
                            severity="warning",
                            suggestion=f"Similar code found in {file2}. Extract to a shared utility."
                        )
                        break

class MagicNumberDetector(BaseCheck):
    """Detect magic numbers and strings"""
    
    def run(self, files):
        for file_path in files:
            if file_path.suffix in ['.js', '.ts', '.vue']:
                lines = self.read_file(file_path)
                
                for i, line in enumerate(lines, 1):
                    # Look for numbers that aren't 0, 1, -1, or 100
                    magic_numbers = re.findall(r'[^.\w](\d{2,})(?!\.\w)', line)
                    for num in magic_numbers:
                        if int(num) not in [0, 1, -1, 100]:
                            self.analyzer.add_issue(
                                check_name=self.__class__.__name__,
                                file_path=file_path,
                                line=i,
                                message=f"Magic number '{num}' detected",
                                severity="info",
                                suggestion="Extract to a named constant"
                            )

class DeepNestingDetector(BaseCheck):
    """Detect deeply nested code (if/for/while)"""
    
    def run(self, files):
        for file_path in files:
            if file_path.suffix in ['.js', '.ts', '.vue']:
                lines = self.read_file(file_path)
                
                nesting_level = 0
                max_nesting = 0
                nesting_lines = []
                
                for i, line in enumerate(lines, 1):
                    # Count opening/closing braces
                    opens = line.count('{') + line.count('if') + line.count('for') + line.count('while')
                    closes = line.count('}')
                    
                    nesting_level += opens - closes
                    
                    if nesting_level > max_nesting:
                        max_nesting = nesting_level
                        nesting_lines = [(i, nesting_level)]
                    elif nesting_level == max_nesting:
                        nesting_lines.append((i, nesting_level))
                
                if max_nesting > 4:
                    for line_no, level in nesting_lines:
                        self.analyzer.add_issue(
                            check_name=self.__class__.__name__,
                            file_path=file_path,
                            line=line_no,
                            message=f"Deep nesting detected (level: {level})",
                            severity="warning",
                            suggestion="Refactor to reduce nesting with early returns or extracted functions"
                        )

class PropMutationDetector(BaseCheck):
    """Detect direct prop mutations in Vue components"""
    
    def run(self, files):
        for file_path in files:
            if file_path.suffix == '.vue':
                lines = self.read_file(file_path)
                
                script_section = False
                props = []
                
                for i, line in enumerate(lines, 1):
                    # Find props definition
                    if 'props:' in line:
                        props_section = True
                    
                    # Collect prop names
                    if props_section:
                        prop_match = re.findall(r'[\'"]([^\'"]+)[\'"]', line)
                        props.extend(prop_match)
                    
                    # Check for mutations in methods
                    if 'methods:' in line:
                        script_section = True
                    
                    if script_section and 'this.' in line:
                        for prop in props:
                            if f'this.{prop} =' in line or f'this.{prop}=' in line:
                                self.analyzer.add_issue(
                                    check_name=self.__class__.__name__,
                                    file_path=file_path,
                                    line=i,
                                    message=f"Direct mutation of prop '{prop}'",
                                    severity="critical",
                                    suggestion="Emit an event to notify parent component of changes"
                                )

class InlineFunctionDetector(BaseCheck):
    """Detect inline functions in templates/JSX"""

                lines = self.read_file(file_path)
                
                patterns = [
                    (r'@click=["\']\([^)]*\)\s*=>', "Inline arrow function in @click", "Move to methods object"),
                    (r'onClick=\{\([^)]*\)\s*=>', "Inline arrow function in onClick", "Extract to named function"),
                ]
                
                for i, line in enumerate(lines, 1):
                    for pattern, message, suggestion in patterns:

class DeadCodeDetector(BaseCheck):
    """Detect potentially dead/unused code"""
    
    def run(self, files):
        # Track function/variable declarations and usage
        declarations = {}
        usages = {}
        
        for file_path in files:
            if file_path.suffix in ['.js', '.ts', '.vue']:
                lines = self.read_file(file_path)
                content = ''.join(lines)
                
                # Find function declarations
                func_decls = re.findall(r'function\s+(\w+)\s*\(', content)
                for func in func_decls:
                    declarations.setdefault(func, []).append(str(file_path))
                
                # Find variable declarations
                var_decls = re.findall(r'(?:let|const|var)\s+(\w+)\s*=', content)
                for var in var_decls:
                    declarations.setdefault(var, []).append(str(file_path))
        
        # Check usage in all files
        for file_path in files:
            if file_path.suffix in ['.js', '.ts', '.vue']:
                lines = self.read_file(file_path)
                content = ''.join(lines)
                
                for name in declarations:
                    if name in content:
                        usages.setdefault(name, []).append(str(file_path))
        
        # Report unused declarations
        for name, declared_in in declarations.items():
            used_in = usages.get(name, [])
            
            # If declared but not used anywhere
            if not used_in:
                for file_path in declared_in:
                    self.analyzer.add_issue(
                        check_name=self.__class__.__name__,
                        file_path=file_path,
                        line=1,
                        message=f"Potentially unused code: '{name}'",
                        severity="info",
                        suggestion="Remove if not needed, or check if it's exported for external use"
                    )

class CircularComponentDetector(BaseCheck):
    """Detect circular component references"""
    
    def run(self, files):
        imports = {}
        
        # Build import map
        for file_path in files:
            if file_path.suffix == '.vue':
                lines = self.read_file(file_path)
                
                for i, line in enumerate(lines, 1):
                    # Look for component imports
                    import_match = re.search(r'import\s+(\w+)\s+from\s+[\'"]([^\'"]+)[\'"]', line)
                    if import_match:
                        component_name = import_match.group(1)
                        import_path = import_match.group(2)
                        
                        imports.setdefault(str(file_path), []).append({
                            'component': component_name,
                            'path': import_path,
                            'line': i
                        })
        
        # Check for circular references (simplified)
        for file_path, deps in imports.items():
            for dep in deps:
                # If dependency is another component
                dep_file = self._resolve_component_path(file_path, dep['path'])
                if dep_file and dep_file in imports:
                    # Check if that component imports back
                    for back_dep in imports.get(dep_file, []):
                        back_file = self._resolve_component_path(dep_file, back_dep['path'])
                        if back_file == file_path:
                            self.analyzer.add_issue(
                                check_name=self.__class__.__name__,
                                file_path=file_path,
                                line=dep['line'],
                                message=f"Circular component reference with {dep['component']}",
                                severity="critical",
                                suggestion="Extract shared functionality to a third component"
                            )
    
    def _resolve_component_path(self, current_file, import_path):
        """Resolve component import path"""
        if import_path.startswith('.'):
            current_dir = os.path.dirname(current_file)
            resolved = os.path.normpath(os.path.join(current_dir, import_path))
            
            # Try with .vue extension
            if not resolved.endswith('.vue'):
                resolved += '.vue'
            
            if os.path.exists(resolved):
                return resolved
        return None

class BundleSizeAnalyzer(BaseCheck):
    """Analyze potential bundle size issues"""
    
    def run(self, files):
        large_files = []
        
        for file_path in files:
            if file_path.suffix in ['.js', '.ts', '.vue']:
                size = file_path.stat().st_size / 1024  # KB
                
                if size > 100:  # Files larger than 100KB
                    large_files.append((file_path, size))
        
        # Report largest files
        large_files.sort(key=lambda x: x[1], reverse=True)
        for file_path, size in large_files[:5]:  # Top 5 largest
            self.analyzer.add_issue(
                check_name=self.__class__.__name__,
                file_path=file_path,
                line=1,
                message=f"Large file detected ({size:.1f}KB)",
                severity="info",
                suggestion="Consider code splitting or lazy loading this component"
            )

class TestCoverageIndicator(BaseCheck):
    """Identify files without corresponding tests"""
    
    def run(self, files):
        # Group files by base name
        source_files = []
        test_files = []
        
        for file_path in files:
            if 'test' in file_path.name.lower() or 'spec' in file_path.name.lower():
                test_files.append(file_path)
            else:
                if file_path.suffix in ['.js', '.ts', '.vue', '.jsx', '.tsx']:
                    source_files.append(file_path)
        
        # Check for missing tests
        for source in source_files:
            base_name = source.stem
            has_test = False
            
            for test in test_files:
                if base_name in test.stem:
                    has_test = True
                    break
            
            if not has_test:
                self.analyzer.add_issue(
                    check_name=self.__class__.__name__,
                    file_path=source,
                    line=1,
                    message="No corresponding test file found",
                    severity="info",
                    suggestion=f"Create a test file for {source.name}"
                )

    analyzer = CodeQualityAnalyzer(
        path=args.path,
        verbose=args.verbose,
        json_output=args.json
    )
    
    analyzer.analyze()

    if analyzer.results["summary"]["critical_issues"] > 0:
        sys.exit(2)  # Critical issues found
    elif analyzer.results["summary"]["warning_issues"] > 0:
        sys.exit(1)  # Warnings found
    else:
        sys.exit(0)  # No issues
